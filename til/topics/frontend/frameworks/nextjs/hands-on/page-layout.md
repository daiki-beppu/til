```tsx
import type { Metadata } from "next";
import { Inter } from "next/font/google";
import "./globals.css";
import { cn } from "@/lib/utils";
import { Button } from "@/components/ui/button";
import Link from "next/link";

const inter = Inter({ subsets: ["latin"] });

export const metadata: Metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

const navItems = [
  { label: "Home", href: "/" },
  { label: "About", href: "/about" },
  { label: "Contact", href: "/contact" },
];

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="ja">
      <body className={cn(inter.className, "min-h-dvh")}>
        <header className="container h-16 flex items-center border-b justify-between">
          <Button asChild variant="ghost">
            <Link href={"/"} className="font-bold ">
              LOGO
            </Link>
          </Button>
          <ul className="flex gap-4">
            {navItems.map((item) => (
              <li key={item.label}>
                <Button asChild variant="ghost">
                  <Link href={item.href}>{item.label}</Link>
                </Button>
              </li>
            ))}
          </ul>
        </header>
        {children}
        <footer className="container h-16 flex items-center border-t sticky top-full">
          <p>&copy; demo-site</p>
        </footer>
      </body>
    </html>
  );
}
```

## 簡単な解説
通常の Link タグだとリンク領域が狭いのでモバイルフレンドリーではない
また、ホバーしたときのフォーカスのスタイルがイケてない
これを解消するために shadcn/ui が用意してくれている Button が有用

動画ではやっていなかったがロゴもボタンにしてHomeに戻るようにしてみた

使い方はめっちゃ簡単で
ターミナルで`bunx --bun shadcn-ui@latest add button`を叩くだけ
もしくは拡張機能を入れると CLI なしでインストールできる

:::message
 shadcn/ui からインストールするといってもなにかパッケージをインストールするとかではなく
実態は shadcn/ui が用意してくれているコードをコピーして貼り付けているだけにすぎない
:::

通常 Button タグ内に Link タグは NG だが  `asChild` を使うことで利用可能にする


## Next.js + Tailwind CSS ｢footerをmainの要素が少ないときでも画面下に固定する方法｣

bodyのスタイルに `min-h-dvh`
footerのスタイルに `sticy` `top-full`
